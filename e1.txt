SELECT nazwa, SUM((`waga`*`ilosc`)*0.7) AS 'netto', SUM((`waga`*`ilosc`)*0.3) AS 'odpad' FROM zasob WHERE `rodzaj`='jedzenie' GROUP BY nazwa;

SELECT * FROM zasob WHERE `rodzaj` IS NULL;
