	idPostac int AUTO_INCREMENT NOT NULL,
	nazwa varchar(40),
	rodzaj ENUM('wiking', 'ptak', 'kobieta'),
	dataUr date,
	wiek int UNSIGNED,
	PRIMARY KEY(idPostac)
);
